#version 300 es precision highp float; uniform float time; uniform vec2 mouse; uniform vec2 resolution; out vec4 outColor; void main(void) { // マウスカーソルの座標を(0.0,0.0)-(1.0,1.0)に変換した2次元ベクトル m を計算 vec2 m = vec2(mouse.x * 2.0 - 1.0, -mouse.y * 2.0 + 1.0); // 画面内の点の座標を(0.0,0.0)-(1.0,1.0)に変換した2次元ベクトル p を計算 vec2 p = (gl_FragCoord.xy * 2.0 - resolution) / min(resolution.x, resolution.y); // マウスカーソルと画面内の点との距離 l を計算 vec2 d = m - p; float th = atan(d.y, d.x); float l = length(d); // 適当にいじる l += sin(th * 5.0 + time * 3.0) * .06; // シマシマにする if (mod(l, 0.2) < 0.05) { outColor = vec4(1.0, 0.8, 0.8, 1.0); } else { outColor = vec4(2.0, 1.0, 1.0, 1.0); } } // GLSLリファレンス https://docs.gl/
